252  for (int xx = kernel_width - 1; xx >= 0; --xx) {
253 
254  // index of input signal, used for checking boundary
255  int image_index_Y = y + (kernel_center_Y - yy);
256  int image_index_X = x + (kernel_center_X - xx);
257 
258  // ignore input samples which are out of bound
259  if (image_index_Y >= 0 && image_index_Y <= image.height && image_index_X >= 0 && image_index_X <= image.width) {
260  new_image(y, x) += image(image_index_Y, image_index_X) * kernel(yy, xx);
